# 示范示例

```c
// 声明结构
typedef struct
{
    long len;      // 长度
    data_t *data;  // 数组指针
} vec_rec, *vec_ptr;

// 创建数组
vec_ptr new_vec(long len)
{
    vec_ptr result = (vec_ptr)malloc(sizeof(vec_rec));
    data_t *data = NULL;
    if (!result)
        return NULL;
    result->len = len;
    if (len > 0)
    {
        data = (data_t *)calloc(len, sizeof(data_t));
        if (!data)
        {
            free((void *)result);
            return NULL;
        }
    }
    result->data = data;
    return result;
}

// 获取数组元素
// 返回值：是否获取成功
// dest: 返回元素的指针
int get_vec_element(vec_ptr v, long index, data_t *dest)
{
    if (index < 0 || index >= v->len)
        return 0;
    *dest = v->data[index];
    return 1;
}

// 获取数组长度
long vec_length(vec_ptr v)
{
    return v->len;
}

// 计算数组内元素之和
#define IDENT 0
#define OP +
void combine1(vec_ptr v, data_t *dest)
{
    long i;
    *dest = IDENT;
    for (i = 0; i < vec_length(v); i++)
    {
        data_t val;
        get_vec_element(v, i, &val);
        *dest = *dest OP val;
    }
}
```

# 开始优化

## 1.开启编译器选项

当开启 `-O1` 或 `-O2`，会明显提升效率，减少每元素周期数

## 2.减少函数调用

将不是每次循环都必要的函数调用拿出来，可以提升效率

```c
void combine2(vec_ptr v, data_t *dest)
{
    long i;
    *dest = IDENT;
    long length = vec_length(v);
    for (i = 0; i < length; i++)
    {
        data_t val;
        get_vec_element(v, i, &val);
        *dest = *dest OP val;
    }
}
```
