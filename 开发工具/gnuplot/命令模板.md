```
# 设置标题
set title "Dcr-Graphic" font ",18" textcolor rgb "black"

# 屏幕显示范围
x0 = -10
x1 = 10
y0 = -2
y1 = 2

set xrange [x0:x1]
set yrange [-2:2]

# 设置绘制范围
set size 1,1
set origin 0,0

# 设置 x, y 轴的刻度线
set xzeroaxis lt 3 lw 2
set yzeroaxis lt 3 lw 2
set xtics 1 nomirror
set ytics 1 nomirror

# 设置 x, y 轴偏移，使其位于屏幕中心
set xtics axis scale -1 nomirror
set ytics axis scale -1 nomirror

# 设置 x, y 的刻度字体大小
set xtics font ", 18"
set ytics font ", 18"

# 不显示原点刻度
set xtics add (" " 0)
set ytics add (" " 0)

# 设置原点图标
# set label "O" at 0.2,0.2 font ",20"
set object 1 circle at 0,0 size 0.1 fillstyle solid border lc rgb "black" fillcolor rgb "black"

# 绘制函数
plot 0.999 ** x
```
